function buildHonor(){
  const el=document.getElementById('tab-honor');if(!el)return;
  const named=getNamedStu(),sc=getScores();
  const days=new Date(YEAR,MONTH,0).getDate();
  const abs={};named.forEach(x=>abs[x.id]=0);
  for(let d=1;d<=days;d++){const a=LS.get(K.att(CLS,YEAR,MONTH,d));if(!a)continue;named.forEach(x=>{if(a[x.id]==='absent')abs[x.id]++;});}

  const ranked=named.map(s=>{
    const v=sc[s.id]||{};
    const score=(v.share||0)+(v.interact||0)+(v.tasks||0)+(v.conduct||0);
    const att=Math.max(0,40-abs[s.id]*2);// نقاط الحضور
    return {...s,score,att,total:score+att};
  }).sort((a,b)=>b.total-a.total).slice(0,10);

  if(!ranked.length){el.innerHTML='<div class="card"><p style="text-align:center;color:var(--muted);padding:20px">لا توجد بيانات بعد</p></div>';return;}

  const top3=ranked.slice(0,3);
  const podiumOrder=[top3[1],top3[0],top3[2]].filter(Boolean);
  const pbClass=['pb2','pb1','pb3'];const pClass=['p2','p1','p3'];

  el.innerHTML=`<div class="card">
    <div class="card-title">🏆 لوحة الشرف — ${CLS} | ${MONTH}/${YEAR}</div>
    <div class="honor-podium">
      ${podiumOrder.map((s,i)=>{
        const p=LS.get(K.img(CLS,s.id));const av=p?`<img src="${p}">`:`<span>${s.name[0]}</span>`;
        return `<div class="podium-place ${pClass[i]}">
          <div class="podium-avatar">${av}</div>
          <div class="podium-name">${s.name.split(' ')[0]}</div>
          <div class="podium-score">${s.total} نقطة</div>
          <div class="podium-block ${pbClass[i]}">${i===1?'🥇':i===0?'🥈':'🥉'}</div>
        </div>`;
      }).join('')}
    </div>
    <div class="honor-list">
      ${ranked.map((s,i)=>{
        const p=LS.get(K.img(CLS,s.id));const av=p?`<img src="${p}">`:`<span>${s.name[0]}</span>`;
        const medals=['🥇','🥈','🥉'];
        return `<div class="honor-row">
          <div class="honor-rank">${medals[i]||i+1}</div>
          <div class="honor-av">${av}</div>
          <div class="honor-info"><div class="honor-name">${s.name}</div><div class="honor-sub">تقييم: ${s.score}/40 | حضور: ${s.att}/40</div></div>
          <div class="honor-score">${s.total}</div>
        </div>`;
      }).join('')}
    </div>
    <p style="font-size:10px;color:var(--muted);margin-top:10px;text-align:center">النقاط = التقييم + نقاط الحضور (40 - غياب×2)</p>
  </div>`;
}

function buildCompare(){

function getGB(id){return LS.get(`ba_gb_${CLS}_${id}`)||{};}
function saveGB(id,data){LS.set(`ba_gb_${CLS}_${id}`,data);}

function gradeLabel(pct){
  if(pct>=90)return{t:'ممتاز',c:'g-a'};
  if(pct>=80)return{t:'جيد جداً',c:'g-b'};
  if(pct>=70)return{t:'جيد',c:'g-b'};
  if(pct>=60)return{t:'مقبول',c:'g-c'};
  if(pct>=50)return{t:'دون المستوى',c:'g-d'};
  return{t:'ضعيف',c:'g-d'};
}

function calcTotal(gb){
  return GB_COLS.reduce((sum,c)=>sum+(parseFloat(gb[c.key])||0),0);
}

function saveGBField(id, key, max, inp){
  const val = parseFloat(inp.value);
  inp.classList.toggle('over', val > max);
  if(val > max){ showToast(`⚠️ الحد الأقصى ${max}`); return; }
  const gb = getGB(id);
  if(inp.value==='') delete gb[key]; else gb[key] = val||0;
  saveGB(id, gb);
  // تحديث المجموع في نفس الصف فوراً
  const row = inp.closest('tr');
  if(row){
    const tot = calcTotal(gb);
    const totEl = row.querySelector('.gb-total');
    const grEl  = row.querySelector('.gb-grade');
    if(totEl) totEl.textContent = tot||'-';
    if(grEl && tot){
      const gl = gradeLabel(tot);
      grEl.textContent = gl.t;
      grEl.className = 'gb-grade '+gl.c;
      totEl.style.color = tot>=90?'var(--green)':tot>=70?'var(--navy)':tot>=50?'var(--orange)':'var(--red)';
    }
  }
  // تحديث الإحصائيات
  refreshGBStats();
}

function refreshGBStats(){
  const named = getNamedStu();
  const totals = named.map(s=>calcTotal(getGB(s.id))).filter(v=>v>0);
  const avg = totals.length ? Math.round(totals.reduce((a,b)=>a+b,0)/totals.length) : 0;
  const pass = totals.filter(v=>v>=50).length;
  const fail = totals.filter(v=>v<50).length;
  const top  = totals.length ? Math.max(...totals) : 0;
  const stats = document.querySelectorAll('.gb-stat .n');
  if(stats[1]) stats[1].textContent = avg||'-';
  if(stats[2]) stats[2].textContent = pass;
  if(stats[3]) stats[3].textContent = fail;
  if(stats[4]) stats[4].textContent = top||'-';
}

function buildDistChart(named){
  const ranges=[
    {label:'90-100',min:90,max:100,color:'var(--green)'},
    {label:'80-89', min:80,max:89, color:'#2980B9'},
    {label:'70-79', min:70,max:79, color:'var(--navy)'},
    {label:'60-69', min:60,max:69, color:'var(--orange)'},
    {label:'50-59', min:50,max:59, color:'#E67E22'},
    {label:'<50',   min:0, max:49, color:'var(--red)'},
  ];
  const counts = ranges.map(r=>{
    const c = named.filter(s=>{const t=calcTotal(getGB(s.id));return t>=r.min&&t<=r.max;}).length;
    return {...r,count:c};
  });
  const maxC = Math.max(...counts.map(x=>x.count),1);
  return `<div style="display:flex;align-items:flex-end;gap:6px;height:80px;padding:0 4px">
    ${counts.map(r=>{
      const h = Math.max(Math.round(r.count/maxC*100),r.count>0?8:2);
      return `<div style="flex:1;display:flex;flex-direction:column;align-items:center;gap:2px">
        <div style="font-size:10px;font-weight:800;color:var(--text)">${r.count||''}</div>
        <div style="width:100%;height:${h}%;border-radius:5px 5px 0 0;background:${r.color};min-height:3px"></div>
        <div style="font-size:8px;font-weight:700;color:var(--muted)">${r.label}</div>
      </div>`;
    }).join('')}
  </div>`;
}

function exportGBExcel(){
  if(typeof XLSX==='undefined'){showToast('⏳ جارٍ التحميل...');return;}
  const named = getNamedStu();
  const headers = ['م','اسم الطالب',...GB_COLS.map(c=>`${c.label} (${c.max})`),'المجموع /100','النسبة%','التقدير'];

  named.forEach((s,i)=>{
    const gb = getGB(s.id);
    const tot = calcTotal(gb);
    const vals = GB_COLS.map(c=>gb[c.key]||0);
    const gl = gradeLabel(tot);
    data.push([i+1, s.name, ...vals, tot, tot+'%', gl.t]);

  const avgs = GB_COLS.map(c=>{
    const vs = named.map(s=>parseFloat(getGB(s.id)[c.key])||0).filter(v=>v>0);
    return vs.length ? Math.round(vs.reduce((a,b)=>a+b,0)/vs.length*10)/10 : 0;
  });
  const allTots = named.map(s=>calcTotal(getGB(s.id))).filter(v=>v>0);
  const avgTot = allTots.length ? Math.round(allTots.reduce((a,b)=>a+b,0)/allTots.length) : 0;
  data.push(['', 'المتوسط', ...avgs, avgTot, avgTot+'%', gradeLabel(avgTot).t]);

function cycleQA(id, skillKey, btn){
  const qa = getQA(id, qaPeriod);
  const cur = qa[skillKey]||'';
  const idx = QA_LEVELS.findIndex(l=>l.key===cur);
  const next = QA_LEVELS[(idx+1) % (QA_LEVELS.length+1)];
  if(!next || idx===QA_LEVELS.length-1){
    // العودة للفارغ
    delete qa[skillKey];
    btn.textContent='—';
    btn.className='qa-btn qa-e';
    btn.dataset.val='';
  } else {
    qa[skillKey]=next.key;
    btn.textContent=next.icon;
    btn.className=`qa-btn ${next.cls} on`;
    btn.dataset.val=next.key;
  }
  saveQA(id, qaPeriod, qa);
  // تحديث التقدير العام في نفس الصف
  const row = btn.closest('tr');
  if(row){
    const ov = getOverallLevel(qa);
    const grEl = row.querySelector('.gb-grade');
    if(grEl){grEl.textContent=ov.label; grEl.className=`gb-grade ${ov.cls}`;}
  }
}

      const lv = QA_LEVELS.find(l=>l.key===v);
      return `<div class="qa-skill-row">
        <div class="qa-skill-name">${sk.icon} ${sk.label}</div>
        <div class="qa-skill-btns">
          ${QA_LEVELS.map(l=>`<button class="qa-btn ${l.cls}${v===l.key?' on':''}"
            onclick="setQA(${s.id},'${sk.key}','${l.key}',this)" title="${l.label}">${l.icon}</button>`).join('')}
          <button class="qa-btn" style="background:var(--surface2);color:var(--muted)"
            onclick="setQA(${s.id},'${sk.key}','',this)" title="مسح">×</button>
        </div>
      </div>`;
    }).join('');
    const noteKey = `ba_qanote_${CLS}_${s.id}_${qaPeriod.replace(/ /g,'_')}`;
    const note = LS.get(noteKey)||'';
    return `<div class="qa-stu-card">
      <div class="qa-stu-hdr">
        ${av}
        <span>${s.name}</span>
        <span class="gb-grade ${overall.cls}" style="margin-right:auto">${overall.label}</span>
      </div>
      <div class="qa-stu-body">
        ${skills}
        <div style="margin-top:10px">
          <div style="font-size:11px;font-weight:700;color:var(--muted);margin-bottom:4px">ملاحظة عامة:</div>
          <textarea class="note-area" style="min-height:50px" placeholder="اكتب ملاحظة للطالب..."
            onchange="LS.set('${noteKey}',this.value)">${note}</textarea>
        </div>
      </div>
    </div>`;
  }).join('');
}

function setQA(id, skillKey, val, btn){
  const qa = getQA(id, qaPeriod);
  if(val) qa[skillKey]=val; else delete qa[skillKey];
  saveQA(id, qaPeriod, qa);
  // تحديث أزرار نفس المهارة
  const skillRow = btn.closest('.qa-skill-row');
  if(skillRow){
    skillRow.querySelectorAll('.qa-btn').forEach(b=>b.classList.remove('on'));
    if(val) btn.classList.add('on');
  }
  // تحديث التقدير العام في البطاقة
  const card = btn.closest('.qa-stu-card');
  if(card){
    const ov = getOverallLevel(qa);
    const grEl = card.querySelector('.gb-grade');
    if(grEl){grEl.textContent=ov.label; grEl.className=`gb-grade ${ov.cls}`;}
  }
}

function getOverallLevel(qa){
  const counts={E:0,G:0,A:0,W:0};
  QA_SKILLS.forEach(sk=>{if(qa[sk.key])counts[qa[sk.key]]=(counts[qa[sk.key]]||0)+1;});
  const total=Object.values(counts).reduce((a,b)=>a+b,0);
  if(!total) return {label:'—', cls:'b-p'};
  if(counts.E/total>=0.6) return {label:'ممتاز 🌟', cls:'g-a'};
  if((counts.E+counts.G)/total>=0.7) return {label:'جيد ✅', cls:'g-b'};
  if(counts.W/total>=0.4) return {label:'يحتاج دعم ❌', cls:'g-d'};
  return {label:'يحتاج تحسين ⚠️', cls:'g-c'};
}

// ── تصدير Excel ──
function exportQAExcel(){
  if(typeof XLSX==='undefined'){showToast('⏳ جارٍ التحميل...');return;}
  const named = getNamedStu();
  const headers = ['م','اسم الطالب',...QA_SKILLS.map(s=>s.label),'التقدير العام'];
  const data = [['الصف:',CLS,'الفصل:',qaPeriod,'المعلم:',CU?CU.name:''],[], headers];
  named.forEach((s,i)=>{
    const qa = getQA(s.id, qaPeriod);
    const ov = getOverallLevel(qa);
    const lvMap={'E':'ممتاز','G':'جيد','A':'يحتاج تحسين','W':'ضعيف','':`—`};
    data.push([i+1, s.name, ...QA_SKILLS.map(sk=>lvMap[qa[sk.key]||'']), ov.label]);
  });
  const wb = XLSX.utils.book_new();
  const ws = XLSX.utils.aoa_to_sheet(data);
  ws['!cols'] = [{wch:4},{wch:24},...Array(QA_SKILLS.length).fill({wch:16}),{wch:16}];
  XLSX.utils.book_append_sheet(wb, ws, qaPeriod);
  XLSX.writeFile(wb, `تقويم_نوعي_${CLS}_${qaPeriod}.xlsx`);
  showToast('✅ تم تصدير التقويم النوعي');
}

// ── تصدير PDF ──
function exportQAPDF(){
  if(typeof window.jspdf==='undefined'){showToast('⏳ جارٍ التحميل...');return;}
  const {jsPDF} = window.jspdf;
  const doc = new jsPDF({orientation:'l', unit:'mm', format:'a4'});
  const named = getNamedStu();
  const lvMap={'E':'ممتاز','G':'جيد','A':'يحتاج تحسين','W':'ضعيف','':`—`};

  doc.setFontSize(14);
  doc.text('Beit Awa Boys Secondary School', 148, 14, {align:'center'});
  doc.setFontSize(11);
  doc.text(`Qualitative Assessment | Class: ${CLS} | ${qaPeriod}`, 148, 22, {align:'center'});
  doc.line(10, 26, 287, 26);

  // رؤوس الأعمدة
  let y=32; doc.setFontSize(7);
  const heads=['No.','Name','Listen','Reading','Poetry','Grammar','Dictation','Handwr.','Expression','Conduct','H.Work','Overall'];
  const xs=[10,20,75,88,100,112,124,136,148,160,171,182];
  heads.forEach((h,i)=>doc.text(h, xs[i], y));
  doc.line(10,y+2,287,y+2); y+=7;

  named.forEach((s,i)=>{
    if(y>195){doc.addPage();y=20;}
    const qa = getQA(s.id, qaPeriod);
    const ov = getOverallLevel(qa);
    const vals = [i+1, s.name.substring(0,12), ...QA_SKILLS.map(sk=>lvMap[qa[sk.key]||'']), ov.label];
    vals.forEach((v,vi)=>doc.text(String(v), xs[vi], y));
    y+=6;
  });
  doc.save(`تقويم_نوعي_${CLS}_${qaPeriod}.pdf`);
  showToast('✅ تم تصدير PDF التقويم النوعي');
}
